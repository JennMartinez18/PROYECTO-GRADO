import mysql.connector
from fastapi import HTTPException
from config.db_config import get_db_connection
from models.Usuarios_Model import Usuario
from fastapi.encoders import jsonable_encoder

class UsuariosController:

    def listar_usuarios(self):
        try:
            conn = get_db_connection()
            cursor = conn.cursor()
            cursor.execute("SELECT * FROM usuarios")
            result = cursor.fetchall()
            payload = []
            content = {}
            for data in result:
                content = {
                    'id': data[0],
                    'nombre': data[1],
                    'apellido': data[2],
                    'email': data[3],
                    'password': data[4],
                    'telefono': data[5],
                    'cedula': data[6],
                    'rol_id': data[7],
                    'activo': data[8],
                    'created_at': data[9],
                    'updated_at': data[10]
                }
                payload.append(content)
                content = {}
            json_data = jsonable_encoder(payload)
            conn.close()
            return {"resultado": json_data}
        except Exception as error:
            return {"resultado": str(error)}

    def insertar_usuario(self, usuario: Usuario):
        try:
            conn = get_db_connection()
            cursor = conn.cursor()
            nombre = usuario.nombre
            apellido = usuario.apellido
            email = usuario.email
            password = usuario.password
            telefono = usuario.telefono
            cedula = usuario.cedula
            rol_id = usuario.rol_id
            activo = usuario.activo
            
            cursor.execute("""
                INSERT INTO usuarios (nombre, apellido, email, password, telefono, cedula, rol_id, activo) 
                VALUES (%s, %s, %s, %s, %s, %s, %s, %s)
            """, (nombre, apellido, email, password, telefono, cedula, rol_id, activo))
            conn.commit()
            conn.close()
            return {"informacion": "Usuario registrado exitosamente"}
        except Exception as error:
            return {"resultado": str(error)}

    def actualizar_usuario(self, id: int, nuevo_usuario: Usuario):
        try:
            conn = get_db_connection()
            cursor = conn.cursor()
            cursor.execute("SELECT id FROM usuarios WHERE id = %s", (id,))
            resultado = cursor.fetchone()
            if not resultado:
                conn.close()
                return {"informacion": "El usuario no se encuentra en la base de datos"}
            
            nombre = nuevo_usuario.nombre
            apellido = nuevo_usuario.apellido
            email = nuevo_usuario.email
            password = nuevo_usuario.password
            telefono = nuevo_usuario.telefono
            cedula = nuevo_usuario.cedula
            rol_id = nuevo_usuario.rol_id
            activo = nuevo_usuario.activo
            
            cursor.execute("""
                UPDATE usuarios SET 
                nombre = %s,
                apellido = %s,
                email = %s,
                password = %s,
                telefono = %s,
                cedula = %s,
                rol_id = %s,
                activo = %s
                WHERE id = %s
            """, (nombre, apellido, email, password, telefono, cedula, rol_id, activo, id))
            conn.commit()
            conn.close()
            return {"informacion": "Usuario actualizado"}
        except Exception as error:
            return {"resultado": str(error)}

    def eliminar_usuario(self, id: int):
        try:
            conn = get_db_connection()
            cursor = conn.cursor()
            cursor.execute("SELECT id FROM usuarios WHERE id = %s", (id,))
            resultado = cursor.fetchone()
            if not resultado:
                conn.close()
                return {"informacion": "El usuario no se encuentra en la base de datos"}

            cursor.execute("DELETE FROM usuarios WHERE id = %s", (id,))
            conn.commit()
            cursor.close()

            cursor = conn.cursor()
            cursor.execute("ALTER TABLE usuarios AUTO_INCREMENT = 1")
            conn.commit()
            conn.close()
            return {"informacion": "Usuario eliminado"}
        except Exception as error:
            return {"resultado": str(error)}

    def obtener_doctores(self):
        """Obtener usuarios con rol de doctor (rol_id 2 o 3)"""
        try:
            conn = get_db_connection()
            cursor = conn.cursor(dictionary=True)
            cursor.execute("""
                SELECT u.*, r.nombre as rol_nombre
                FROM usuarios u
                JOIN roles r ON u.rol_id = r.id
                WHERE u.rol_id IN (2, 3) AND u.activo = 1
                ORDER BY u.nombre, u.apellido
            """)
            result = cursor.fetchall()
            # Remover contraseñas
            for user in result:
                user.pop('password', None)
            conn.close()
            return {"resultado": result}
        except Exception as error:
            return {"resultado": str(error)}
